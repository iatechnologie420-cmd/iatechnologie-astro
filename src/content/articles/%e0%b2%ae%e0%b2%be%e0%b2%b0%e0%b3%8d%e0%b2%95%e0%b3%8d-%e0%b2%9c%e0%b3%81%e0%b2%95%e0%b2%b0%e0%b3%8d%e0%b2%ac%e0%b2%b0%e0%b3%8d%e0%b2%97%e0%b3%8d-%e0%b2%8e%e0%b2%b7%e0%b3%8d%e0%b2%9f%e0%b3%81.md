<p>ಹಲವಾರು ವರ್ಷಗಳಿಂದ, ಫೇಸ್‌ಬುಕ್ ಸಂಸ್ಥಾಪಕ ಮಾರ್ಕ್ ಜುಕರ್‌ಬರ್ಗ್ ಅವರ ಎತ್ತರದ ಪ್ರಶ್ನೆಯು ಹೆಚ್ಚಿನ ಊಹಾಪೋಹ ಮತ್ತು ಚರ್ಚೆಯ ವಿಷಯವಾಗಿದೆ. ಕೆಲವರು ಇದು ತೋರುತ್ತಿರುವುದಕ್ಕಿಂತ ಚಿಕ್ಕದಾಗಿದೆ ಎಂದು ಹೇಳಿದರೆ, ಇತರರು ಇದು ನಿಜವಾಗಿಯೂ ಸಾಕಷ್ಟು ಎತ್ತರವಾಗಿದೆ ಎಂದು ಹೇಳುತ್ತಾರೆ. ಹಾಗಾದರೆ ಮಾರ್ಕ್ ಜುಕರ್‌ಬರ್ಗ್ ಅವರ ಎತ್ತರದ ಬಗ್ಗೆ ಸತ್ಯವೇನು?</p>



<h4 class="wp-block-heading"><span class="ez-toc-section" id="%E0%B2%86%E0%B2%A7%E0%B2%BE%E0%B2%B0%E0%B2%B0%E0%B2%B9%E0%B2%BF%E0%B2%A4_%E0%B2%B5%E0%B2%A6%E0%B2%82%E0%B2%A4%E0%B2%BF%E0%B2%97%E0%B2%B3%E0%B3%81"></span>ಆಧಾರರಹಿತ ವದಂತಿಗಳು<span class="ez-toc-section-end"></span></h4>



<p>ಮಾರ್ಕ್ ಜುಕರ್‌ಬರ್ಗ್ 1 ಮೀ 60 ಕ್ಕಿಂತ ಕಡಿಮೆ ಎತ್ತರವಿದೆ ಎಂದು ಇಂಟರ್ನೆಟ್‌ನಲ್ಲಿ ಅನೇಕ ವದಂತಿಗಳು ಹರಿದಾಡುತ್ತಿವೆ. ಇತರ ಸೆಲೆಬ್ರಿಟಿಗಳೊಂದಿಗೆ ಫೋಟೋ ಹೋಲಿಕೆಗಳು ಮತ್ತು ಮಾರ್ಕ್ ಹೆಚ್ಚಾಗಿ ಜೋಲಾಡುವ ಬಟ್ಟೆ ಮತ್ತು ಹೆಡೆಗಳನ್ನು ಧರಿಸಿರುವುದು ಈ ವದಂತಿಗಳಿಗೆ ಉತ್ತೇಜನ ನೀಡಿದೆ. ಆದಾಗ್ಯೂ, ಈ ಹಕ್ಕುಗಳು ಸಂಪೂರ್ಣವಾಗಿ ಆಧಾರರಹಿತವಾಗಿವೆ ಮತ್ತು ಯಾವುದೇ ಕಾಂಕ್ರೀಟ್ ಪುರಾವೆಗಳನ್ನು ಆಧರಿಸಿಲ್ಲ ಎಂಬುದನ್ನು ಎತ್ತಿ ತೋರಿಸುವುದು ಮುಖ್ಯವಾಗಿದೆ.</p>



<h4 class="wp-block-heading"><span class="ez-toc-section" id="%E0%B2%B5%E0%B2%BE%E0%B2%B8%E0%B3%8D%E0%B2%A4%E0%B2%B5_%E0%B2%AE%E0%B2%BE%E0%B2%B0%E0%B3%8D%E0%B2%95%E0%B3%8D_%E0%B2%9C%E0%B3%81%E0%B2%95%E0%B2%B0%E0%B3%8D%E2%80%8C%E0%B2%AC%E0%B2%B0%E0%B3%8D%E0%B2%97%E0%B3%8D_%E0%B2%85%E0%B2%B5%E0%B2%B0_%E0%B2%A8%E0%B2%BF%E0%B2%9C%E0%B2%B5%E0%B2%BE%E0%B2%A6_%E0%B2%8E%E0%B2%A4%E0%B3%8D%E0%B2%A4%E0%B2%B0"></span>ವಾಸ್ತವ: ಮಾರ್ಕ್ ಜುಕರ್‌ಬರ್ಗ್ ಅವರ ನಿಜವಾದ ಎತ್ತರ<span class="ez-toc-section-end"></span></h4>



<p>ಮಾರ್ಕ್ ಜುಕರ್‌ಬರ್ಗ್ ಅವರ ನಿಜವಾದ ಎತ್ತರವು ಚರ್ಚಿಸಲು ಕಷ್ಟಕರವಾದ ವಿಷಯವಾಗಿ ಉಳಿದಿದೆ, ಏಕೆಂದರೆ ವಾಣಿಜ್ಯೋದ್ಯಮಿ ತನ್ನ ಖಾಸಗಿ ಜೀವನದ ಬಗ್ಗೆ ವಿವೇಚನೆಯಿಂದ ಇರಲು ಬಯಸುತ್ತಾನೆ ಮತ್ತು ಸಾರ್ವಜನಿಕವಾಗಿ ತನ್ನ ಎತ್ತರವನ್ನು ಎಂದಿಗೂ ತಿಳಿಸಲಿಲ್ಲ. ಆದ್ದರಿಂದ ಅವರ ಎತ್ತರಕ್ಕೆ ಸಂಬಂಧಿಸಿದ ಮಾಹಿತಿಯು ಕಡಿಮೆ ಮತ್ತು ಪರಿಶೀಲಿಸಲು ಕಷ್ಟ, ಆದರೆ ಮೂಲಗಳ ಅಡ್ಡ-ಉಲ್ಲೇಖದ ಪ್ರಕಾರ, ಮಾರ್ಕ್ ಜುಕರ್‌ಬರ್ಗ್ 1 ಮೀ 70 ಎತ್ತರವಿದೆ.</p>

<h3 class="wp-block-heading"><span class="ez-toc-section" id="%E0%B2%AE%E0%B2%BE%E0%B2%B0%E0%B3%8D%E0%B2%95%E0%B3%8D_%E0%B2%9C%E0%B3%81%E0%B2%95%E0%B2%B0%E0%B3%8D%E2%80%8C%E0%B2%AC%E0%B2%B0%E0%B3%8D%E0%B2%97%E0%B3%8D_%E0%B2%85%E0%B2%B5%E0%B2%B0_%E0%B2%A4%E0%B3%82%E0%B2%95"></span>ಮಾರ್ಕ್ ಜುಕರ್‌ಬರ್ಗ್ ಅವರ ತೂಕ<span class="ez-toc-section-end"></span></h3>



<p>ನಾವು &#8220;ಮಾರ್ಕ್ ಜುಕರ್‌ಬರ್ಗ್‌ನ ತೂಕ&#8221; ದ ಬಗ್ಗೆ ಮಾತನಾಡುವಾಗ, ಅದರ ಪ್ರಾಮುಖ್ಯತೆಯನ್ನು ಉಲ್ಲೇಖಿಸುವುದು ಮಾತ್ರವಲ್ಲ, ಹೆಚ್ಚು ಅಕ್ಷರಶಃ ಪ್ರಶ್ನೆಯನ್ನು ನೋಡುವುದು: ಅವರ ದೈಹಿಕ ತೂಕ: ಈ ಲೇಖನವನ್ನು ಬರೆಯುವ ಅವರು ಗಂಟೆಗೆ ಸುಮಾರು 70 ಕೆಜಿ ತೂಗುತ್ತಾರೆ.</p>



<h4 class="wp-block-heading"><span class="ez-toc-section" id="%E0%B2%B5%E0%B2%BF%E0%B2%95%E0%B2%B8%E0%B2%A8%E0%B2%97%E0%B3%8A%E0%B2%82%E0%B2%A1_%E0%B2%B8%E0%B2%BF%E0%B2%B2%E0%B3%82%E0%B2%AF%E0%B3%86%E0%B2%9F%E0%B3%8D"></span>ವಿಕಸನಗೊಂಡ ಸಿಲೂಯೆಟ್<span class="ez-toc-section-end"></span></h4>



<p>ವರ್ಷಗಳಲ್ಲಿ, ಮಾರ್ಕ್ ಜುಕರ್‌ಬರ್ಗ್ ಗಮನಾರ್ಹ ಭೌತಿಕ ವಿಕಸನಕ್ಕೆ ಒಳಗಾಗಿದ್ದಾರೆ. ವಾಸ್ತವವಾಗಿ, ಅವಳ ಸಿಲೂಯೆಟ್ ಕಾಲಾನಂತರದಲ್ಲಿ ರೂಪಾಂತರಗೊಂಡಿದೆ ಎಂದು ಎಲ್ಲರಿಗೂ ತಿಳಿದಿದೆ. ಫೇಸ್‌ಬುಕ್‌ನ ಆರಂಭಿಕ ವರ್ಷಗಳಲ್ಲಿ, ಅವರು ಸಾಮಾನ್ಯವಾಗಿ ತೆಳ್ಳಗಿನ, ದುರ್ಬಲ ಯುವಕನಂತೆ ಕಾಣುತ್ತಿದ್ದರು. ಆದಾಗ್ಯೂ, ಕಾಲಾನಂತರದಲ್ಲಿ, ಅವರು ಸ್ನಾಯುವಿನ ದ್ರವ್ಯರಾಶಿಯನ್ನು ಪಡೆದರು ಮತ್ತು ಹೆಚ್ಚು ಅಥ್ಲೆಟಿಕ್ ನೋಟವನ್ನು ಅಳವಡಿಸಿಕೊಂಡರು.</p>



<h4 class="wp-block-heading"><span class="ez-toc-section" id="%E0%B2%AE%E0%B2%BE%E0%B2%B0%E0%B3%8D%E0%B2%95%E0%B3%8D_%E0%B2%9C%E0%B3%81%E0%B2%95%E0%B2%B0%E0%B3%8D%E2%80%8C%E0%B2%AC%E0%B2%B0%E0%B3%8D%E0%B2%97%E0%B3%8D_%E0%B2%85%E0%B2%B5%E0%B2%B0_%E0%B2%86%E0%B2%B9%E0%B2%BE%E0%B2%B0%E0%B2%95%E0%B3%8D%E0%B2%B0%E0%B2%AE"></span>ಮಾರ್ಕ್ ಜುಕರ್‌ಬರ್ಗ್ ಅವರ ಆಹಾರಕ್ರಮ<span class="ez-toc-section-end"></span></h4>



<p>ಈ ದೈಹಿಕ ಬದಲಾವಣೆಗೆ ಒಂದು ಸಂಭವನೀಯ ಕಾರಣವೆಂದರೆ ಮಾರ್ಕ್ ಜುಕರ್‌ಬರ್ಗ್ ಅನುಸರಿಸಿದ ಆಹಾರಕ್ರಮ. ಕೆಲವು ಮೂಲಗಳ ಪ್ರಕಾರ, ಅವರು ಸಸ್ಯಾಹಾರಿ ಅಥವಾ ಸಸ್ಯಾಹಾರಿ ಆಹಾರವನ್ನು ಅಳವಡಿಸಿಕೊಂಡರು. ಹಣ್ಣುಗಳು, ತರಕಾರಿಗಳು, ಧಾನ್ಯಗಳು ಮತ್ತು ಸಸ್ಯ ಪ್ರೋಟೀನ್‌ಗಳ ವಿಶೇಷ ಸೇವನೆಯ ಆಧಾರದ ಮೇಲೆ ಈ ಆಹಾರದ ಆಯ್ಕೆಯು ಉತ್ತಮ ಆರೋಗ್ಯ ಮತ್ತು ತೂಕ ನಷ್ಟಕ್ಕೆ ಕೊಡುಗೆ ನೀಡುತ್ತದೆ.</p>



<h4 class="wp-block-heading"><span class="ez-toc-section" id="%E0%B2%AE%E0%B2%BE%E0%B2%B0%E0%B3%8D%E0%B2%95%E0%B3%8D_%E0%B2%9C%E0%B3%81%E0%B2%95%E0%B2%B0%E0%B3%8D%E2%80%8C%E0%B2%AC%E0%B2%B0%E0%B3%8D%E0%B2%97%E0%B3%8D_%E0%B2%85%E0%B2%B5%E0%B2%B0_%E0%B2%A6%E0%B3%88%E0%B2%B9%E0%B2%BF%E0%B2%95_%E0%B2%B5%E0%B3%8D%E0%B2%AF%E0%B2%BE%E0%B2%AF%E0%B2%BE%E0%B2%AE"></span>ಮಾರ್ಕ್ ಜುಕರ್‌ಬರ್ಗ್ ಅವರ ದೈಹಿಕ ವ್ಯಾಯಾಮ<span class="ez-toc-section-end"></span></h4>



<p>ಆರೋಗ್ಯಕರ ಆಹಾರದ ಜೊತೆಗೆ, ಮಾರ್ಕ್ ಜುಕರ್‌ಬರ್ಗ್ ದೈಹಿಕ ವ್ಯಾಯಾಮಕ್ಕೆ ಹೆಚ್ಚಿನ ಪ್ರಾಮುಖ್ಯತೆಯನ್ನು ನೀಡುತ್ತಾರೆ ಎಂದು ತಿಳಿದಿದೆ. ಅವರು ನಿಯಮಿತವಾಗಿ ಓಟ, ದೇಹದಾರ್ಢ್ಯ ಮತ್ತು ಬಾಕ್ಸಿಂಗ್‌ನಂತಹ ಕ್ರೀಡಾ ಚಟುವಟಿಕೆಗಳಲ್ಲಿ ತೊಡಗುತ್ತಾರೆ. ಈ ಚಟುವಟಿಕೆಗಳು ಫಿಟ್ ದೇಹವನ್ನು ಕಾಪಾಡಿಕೊಳ್ಳಲು ಮತ್ತು ಹೆಚ್ಚುವರಿ ಕ್ಯಾಲೊರಿಗಳನ್ನು ಸುಡಲು ಅನುವು ಮಾಡಿಕೊಡುತ್ತದೆ.</p>



<h4 class="wp-block-heading"><span class="ez-toc-section" id="%E0%B2%B5%E0%B3%83%E0%B2%A4%E0%B3%8D%E0%B2%A4%E0%B2%BF%E0%B2%AA%E0%B2%B0_%E0%B2%AE%E0%B2%A4%E0%B3%8D%E0%B2%A4%E0%B3%81_%E0%B2%B5%E0%B3%88%E0%B2%AF%E0%B2%95%E0%B3%8D%E0%B2%A4%E0%B2%BF%E0%B2%95_%E0%B2%9C%E0%B3%80%E0%B2%B5%E0%B2%A8%E0%B2%A6_%E0%B2%A8%E0%B2%A1%E0%B3%81%E0%B2%B5%E0%B2%BF%E0%B2%A8_%E0%B2%B8%E0%B2%AE%E0%B2%A4%E0%B3%8B%E0%B2%B2%E0%B2%A8"></span>ವೃತ್ತಿಪರ ಮತ್ತು ವೈಯಕ್ತಿಕ ಜೀವನದ ನಡುವಿನ ಸಮತೋಲನ<span class="ez-toc-section-end"></span></h4>



<p>ಮಾರ್ಕ್ ಜುಕರ್‌ಬರ್ಗ್ ಅವರ ತೂಕವನ್ನು ವೃತ್ತಿಪರ ಮತ್ತು ವೈಯಕ್ತಿಕ ಜೀವನದ ನಡುವಿನ ಸಮತೋಲಿತ ನಿರ್ವಹಣೆಗೆ ಸಹ ಜೋಡಿಸಬಹುದು. ವಾಸ್ತವವಾಗಿ, ತಮ್ಮ ವೃತ್ತಿಜೀವನದಲ್ಲಿ ಯಶಸ್ವಿಯಾದ ಜನರು ಕೆಲವೊಮ್ಮೆ ತಮ್ಮ ದೈಹಿಕ ಆರೋಗ್ಯವನ್ನು ನಿರ್ಲಕ್ಷಿಸಬಹುದು ಎಂದು ಒತ್ತಿಹೇಳುವುದು ಮುಖ್ಯ. </p>



<p>ಆದಾಗ್ಯೂ, ಮಾರ್ಕ್ ಜುಕರ್‌ಬರ್ಗ್ ತಮ್ಮ ಕೆಲಸ ಮತ್ತು ಅವರ ಆರೋಗ್ಯಕ್ಕೆ ಸಮಾನ ಪ್ರಾಮುಖ್ಯತೆಯನ್ನು ನೀಡುವ ಮೂಲಕ ಈ ಸಮತೋಲನವನ್ನು ಕಾಪಾಡಿಕೊಳ್ಳಲು ಶ್ರಮಿಸಿದ್ದಾರೆ.</p>



<h4 class="wp-block-heading"><span class="ez-toc-section" id="%E0%B2%B8%E0%B2%B5%E0%B2%BE%E0%B2%B2%E0%B3%81%E0%B2%97%E0%B2%B3%E0%B3%81_%E0%B2%85%E0%B2%A6%E0%B2%B0_%E0%B2%95%E0%B3%81%E0%B2%96%E0%B3%8D%E0%B2%AF%E0%B2%BE%E0%B2%A4%E0%B2%BF%E0%B2%97%E0%B3%86_%E0%B2%B8%E0%B2%82%E0%B2%AC%E0%B2%82%E0%B2%A7%E0%B2%BF%E0%B2%B8%E0%B2%BF%E0%B2%B5%E0%B3%86"></span>ಸವಾಲುಗಳು ಅದರ ಕುಖ್ಯಾತಿಗೆ ಸಂಬಂಧಿಸಿವೆ<span class="ez-toc-section-end"></span></h4>



<p>ಮಾರ್ಕ್ ಜುಕರ್‌ಬರ್ಗ್ ಅವರ ಯಶಸ್ಸು ಮತ್ತು ಪ್ರಭಾವಕ್ಕಾಗಿ ಆಗಾಗ್ಗೆ ಮೆಚ್ಚುಗೆ ಪಡೆದರೂ, ಅವರ ಖ್ಯಾತಿಯು ದೈಹಿಕವಾಗಿಯೂ ಸಹ ಹೊರೆಯಾಗಬಹುದು. ನಿರಂತರವಾಗಿ ಜನಮನದಲ್ಲಿರುವುದು ಮತ್ತು ಹೆಚ್ಚಿದ ಮಾಧ್ಯಮದ ಒತ್ತಡವನ್ನು ನಿಭಾಯಿಸುವುದು ವ್ಯಕ್ತಿಯ ಆರೋಗ್ಯ ಮತ್ತು ಯೋಗಕ್ಷೇಮದ ಮೇಲೆ ಪರಿಣಾಮ ಬೀರಬಹುದು.</p>



<p>ತನ್ನ ಆಹಾರದ ಬಗ್ಗೆ ಕಾಳಜಿ ವಹಿಸುವ ಮೂಲಕ, ದೈಹಿಕ ವ್ಯಾಯಾಮದಲ್ಲಿ ತೊಡಗಿಸಿಕೊಳ್ಳುವ ಮೂಲಕ ಮತ್ತು ಕೆಲಸ ಮತ್ತು ವೈಯಕ್ತಿಕ ಜೀವನವನ್ನು ಸಮತೋಲನಗೊಳಿಸುವುದರಿಂದ, ಅವರು ಉತ್ತಮ ದೈಹಿಕ ಸ್ಥಿತಿಯನ್ನು ಕಾಪಾಡಿಕೊಳ್ಳಲು ಸೂತ್ರವನ್ನು ಕಂಡುಕೊಂಡಿದ್ದಾರೆ. ಅವರ ಸ್ಪೂರ್ತಿದಾಯಕ ಪ್ರಯಾಣವು ನಮ್ಮ ಸ್ಥಿತಿ ಅಥವಾ ವೃತ್ತಿಜೀವನವನ್ನು ಲೆಕ್ಕಿಸದೆ ನಮ್ಮ ಆರೋಗ್ಯವನ್ನು ನೋಡಿಕೊಳ್ಳಲು ನಮ್ಮಲ್ಲಿ ಪ್ರತಿಯೊಬ್ಬರಿಗೂ ಪ್ರೇರಣೆಯಾಗಿ ಕಾರ್ಯನಿರ್ವಹಿಸುತ್ತದೆ.</p>
